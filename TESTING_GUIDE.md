### Adding More Tests

To add tests for other controllers:

```javascript
// src/__tests__/[controller].test.js
const { controllerFunction } = require("../controllers/[controller]");
const modelMock = require("../models/[model]");

jest.mock("../models/[model]");

describe("[Controller Name]", () => {
  let req, res;

  beforeEach(() => {
    req = { body: {}, user: { id: "test" }, params: {} };
    res = {
      status: jest.fn().mockReturnThis(),
      json: jest.fn().mockReturnThis(),
    };
    jest.clearAllMocks();
  });

  it("should test behavior", async () => {
    await controllerFunction(req, res);
    expect(res.status).toHaveBeenCalledWith(200);
  });
});
```

### Troubleshooting

**Tests timing out?**

- Increase timeout in jest.config.js: `testTimeout: 15000`

**Mock issues?**

- Ensure mocks are defined before imports
- Use `jest.mock()` at the top of test file
- Call `jest.clearAllMocks()` in `beforeEach`

**Environment variable errors?**

- Check src/**tests**/setup.js
- Verify variables are set before controller tests run
